: << 'EOF'
set CFLAGS=%CFLAGS% opengl32.lib glu32.lib
goto :eof
EOF

if [ "$(uname)" = "Darwin" ]; then
  if [ "$DITHER_2D_USE_OPENGL" -eq 1 ]; then
    CFLAGS="$CFLAGS -framework OpenGL"
  fi
  if [ "$DITHER_2D_USE_COREGRAPHICS" -eq 1 ]; then
    CFLAGS="$CFLAGS -framework CoreGraphics"
  fi
else
  CFLAGS="$CFLAGS -lGLEW -lGL -lGLU"
fi
